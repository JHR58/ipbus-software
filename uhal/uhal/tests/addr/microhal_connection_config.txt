* If true, all IPbus packets are routed directly to the devices using the DeviceName to IPaddress map
* futher down - i.e. there is no intermediary ControlHub; this mode is useful for benchtop testing, etc.
* If false, packets are routed to the appropriate ControlHub, as according to either the DeviceName to
* ControlHub mapping below, or as according to the IPbus Name Server (see "use_ipbus_name_server" option).
use_direct_access_mode = false | true

* If set to false, MicroHAL will use a local mapping of DeviceName to ControlHub (see further down)
* If set to true, MicroHAL will query an IPbus Name Server (address specified below) to get a globally
* maintained mapping of DeviceName to ControlHub.
* Note: If use_direct_access_mode = true, then this setting is ignored.
use_ipbus_name_server = false | true

* The addresses of the primary/backup IPbus Name Server services.
ipbus_name_server_addr_primary = "http://ipbusNameServer1.cms:10203"
ipbus_name_server_addr_backup = "http://ipbusNameServer2.cms:10203"

*   1)  use_direct_access_mode = true
*   2)  use_direct_access_mode = false   AND   use_ipbus_name_server = false
*
---------------------------------------------------------------------------------------------------------------------
        DeviceName          |  Locally Defined ControlHub Mapping   |    DirectAccessIPaddr    |   DeviceProtocol
---------------------------------------------------------------------------------------------------------------------
